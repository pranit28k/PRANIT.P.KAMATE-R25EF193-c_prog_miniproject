#define WIDTH 80
#define HEIGHT 25
#define MAX_OBJECTS 100

char canvas[HEIGHT][WIDTH];

/* ============================
   Object Definitions
   ============================ */

typedef enum {
    LINE,
    RECTANGLE,
    CIRCLE,
    TRIANGLE
} ShapeType;

typedef struct {
    int x1, y1;
    int x2, y2;
    int x3, y3;
    int radius;
    char drawChar;
    ShapeType type;
} Shape;

Shape objects[MAX_OBJECTS];
int objectCount = 0;

/* ============================
   Canvas Functions
   ============================ */

void clearCanvas() {
    for (int y = 0; y < HEIGHT; y++)
        for (int x = 0; x < WIDTH; x++)
            canvas[y][x] = ' ';
}

void setPixel(int x, int y, char c) {
    if (x >= 0 && x < WIDTH &&
        y >= 0 && y < HEIGHT)
        canvas[y][x] = c;
}

void displayCanvas() {
    for (int y = 0; y < HEIGHT; y++) {
        for (int x = 0; x < WIDTH; x++)
            putchar(canvas[y][x]);
        putchar('\n');
    }
}

/* ============================
   Drawing Algorithms
   ============================ */

void drawLine(int x1, int y1, int x2, int y2, char c) {
    int dx = abs(x2 - x1);
    int dy = abs(y2 - y1);

    int sx = (x1 < x2) ? 1 : -1;
    int sy = (y1 < y2) ? 1 : -1;

    int err = dx - dy;

    while (1) {
        setPixel(x1, y1, c);

        if (x1 == x2 && y1 == y2)
            break;

        int e2 = 2 * err;

        if (e2 > -dy) {
            err -= dy;
            x1 += sx;
        }

        if (e2 < dx) {
            err += dx;
            y1 += sy;
        }
    }
}

void drawRectangle(int x1, int y1, int x2, int y2, char c) {
    drawLine(x1, y1, x2, y1, c);
    drawLine(x2, y1, x2, y2, c);
    drawLine(x2, y2, x1, y2, c);
    drawLine(x1, y2, x1, y1, c);
}

void drawCircle(int xc, int yc, int r, char c) {
    int x = 0;
    int y = r;
    int d = 3 - 2 * r;

    while (y >= x) {

        setPixel(xc + x, yc + y, c);
        setPixel(xc - x, yc + y, c);
        setPixel(xc + x, yc - y, c);
        setPixel(xc - x, yc - y, c);

        setPixel(xc + y, yc + x, c);
        setPixel(xc - y, yc + x, c);
        setPixel(xc + y, yc - x, c);
        setPixel(xc - y, yc - x, c);

        x++;

        if (d > 0) {
            y--;
            d += 4 * (x - y) + 10;
        } else {
            d += 4 * x + 6;
        }
    }
}

void drawTriangle(
    int x1, int y1,
    int x2, int y2,
    int x3, int y3,
    char c)
{
    drawLine(x1, y1, x2, y2, c);
    drawLine(x2, y2, x3, y3, c);
    drawLine(x3, y3, x1, y1, c);
}

/* ============================
   Object Management
   ============================ */

void addObject(Shape shape) {
    if (objectCount < MAX_OBJECTS)
        objects[objectCount++] = shape;
}

void deleteObject(int index) {
    if (index < 0 || index >= objectCount)
        return;

    for (int i = index; i < objectCount - 1; i++)
        objects[i] = objects[i + 1];

    objectCount--;
}

void modifyObject(int index, Shape newShape) {
    if (index < 0 || index >= objectCount)
        return;

    objects[index] = newShape;
}

/* ============================
   Redraw Entire Scene
   ============================ */

void redrawScene() {
    clearCanvas();

    for (int i = 0; i < objectCount; i++) {

        Shape *s = &objects[i];

        switch (s->type) {

        case LINE:
            drawLine(
                s->x1, s->y1,
                s->x2, s->y2,
                s->drawChar);
            break;

        case RECTANGLE:
            drawRectangle(
                s->x1, s->y1,
                s->x2, s->y2,
                s->drawChar);
            break;

        case CIRCLE:
            drawCircle(
                s->x1, s->y1,
                s->radius,
                s->drawChar);
            break;

        case TRIANGLE:
            drawTriangle(
                s->x1, s->y1,
                s->x2, s->y2,
                s->x3, s->y3,
                s->drawChar);
            break;
        }
    }
}

  #include <stdio.h>
#include <stdlib.h>

#define ROWS 20
#define COLS 50

char screen[ROWS][COLS];

/* Initialize canvas */
void initialize()
{
    int i, j;

    for(i = 0; i < ROWS; i++)
    {
        for(j = 0; j < COLS; j++)
        {
            screen[i][j] = '.';
        }
    }
}

/* Safe plotting */
void plot(int r, int c)
{
    if(r >= 0 && r < ROWS &&
       c >= 0 && c < COLS)
    {
        screen[r][c] = '*';
    }
}

/* Display picture */
void showPicture()
{
    int i, j;

    printf("\n");

    for(i = 0; i < ROWS; i++)
    {
        for(j = 0; j < COLS; j++)
        {
            printf("%c", screen[i][j]);
        }
        printf("\n");
    }
}

/* Draw line using DDA */
void drawLine(int r1, int c1, int r2, int c2)
{
    int dx = c2 - c1;
    int dy = r2 - r1;

    int steps = abs(dx) > abs(dy) ? abs(dx) : abs(dy);

    float xInc = dx / (float)steps;
    float yInc = dy / (float)steps;

    float x = c1;
    float y = r1;

    for(int i = 0; i <= steps; i++)
    {
        plot((int)y, (int)x);

        x += xInc;
        y += yInc;
    }
}

/* Draw rectangle */
void drawRectangle(int row, int col, int width, int height)
{
    drawLine(row, col, row, col + width - 1);
    drawLine(row, col, row + height - 1, col);

    drawLine(row + height - 1,
             col,
             row + height - 1,
             col + width - 1);

    drawLine(row,
             col + width - 1,
             row + height - 1,
             col + width - 1);
}

/* Draw triangle */
void drawTriangle(int row, int col, int height)
{
    drawLine(row, col, row + height, col - height);
    drawLine(row, col, row + height, col + height);

    drawLine(row + height,
             col - height,
             row + height,
             col + height);
}

/* Draw circle */
void drawCircle(int centerRow,
                int centerCol,
                int radius)
{
    for(int r = 0; r < ROWS; r++)
    {
        for(int c = 0; c < COLS; c++)
        {
            int value =
                (r - centerRow) * (r - centerRow) +
                (c - centerCol) * (c - centerCol);

            if(abs(value - radius * radius) <= radius)
            {
                plot(r, c);
            }
        }
    }
}

/* Clear canvas */
void clearPicture()
{
    initialize();
}
