Write functions to draw a circle,
 rectangle, line, and triangle. 
 You should be able to add objects to
  the picture, delete objects from the picture, 
  and modify objects in the picture. 
  Use a 2d array of characters to store the picture.
   Write a function to display the picture
    Additionally, you can use the ncurses package 
    to create menus and draw pictures.

    RESPONSE 1:
    #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>

#define WIDTH 80
#define HEIGHT 25
#define MAX_OBJECTS 100

char canvas[HEIGHT][WIDTH];

/* ============================
   Object Definitions
   ============================ */

typedef enum {
    LINE,
    RECTANGLE,
    CIRCLE,
    TRIANGLE
} ShapeType;

typedef struct {
    int x1, y1;
    int x2, y2;
    int x3, y3;
    int radius;
    char drawChar;
    ShapeType type;
} Shape;

Shape objects[MAX_OBJECTS];
int objectCount = 0;

/* ============================
   Canvas Functions
   ============================ */

void clearCanvas() {
    for (int y = 0; y < HEIGHT; y++)
        for (int x = 0; x < WIDTH; x++)
            canvas[y][x] = ' ';
}

void setPixel(int x, int y, char c) {
    if (x >= 0 && x < WIDTH &&
        y >= 0 && y < HEIGHT)
        canvas[y][x] = c;
}

void displayCanvas() {
    for (int y = 0; y < HEIGHT; y++) {
        for (int x = 0; x < WIDTH; x++)
            putchar(canvas[y][x]);
        putchar('\n');
    }
}

/* ============================
   Drawing Algorithms
   ============================ */

void drawLine(int x1, int y1, int x2, int y2, char c) {
    int dx = abs(x2 - x1);
    int dy = abs(y2 - y1);

    int sx = (x1 < x2) ? 1 : -1;
    int sy = (y1 < y2) ? 1 : -1;

    int err = dx - dy;

    while (1) {
        setPixel(x1, y1, c);

        if (x1 == x2 && y1 == y2)
            break;

        int e2 = 2 * err;

        if (e2 > -dy) {
            err -= dy;
            x1 += sx;
        }

        if (e2 < dx) {
            err += dx;
            y1 += sy;
        }
    }
}

void drawRectangle(int x1, int y1, int x2, int y2, char c) {
    drawLine(x1, y1, x2, y1, c);
    drawLine(x2, y1, x2, y2, c);
    drawLine(x2, y2, x1, y2, c);
    drawLine(x1, y2, x1, y1, c);
}

void drawCircle(int xc, int yc, int r, char c) {
    int x = 0;
    int y = r;
    int d = 3 - 2 * r;

    while (y >= x) {

        setPixel(xc + x, yc + y, c);
        setPixel(xc - x, yc + y, c);
        setPixel(xc + x, yc - y, c);
        setPixel(xc - x, yc - y, c);

        setPixel(xc + y, yc + x, c);
        setPixel(xc - y, yc + x, c);
        setPixel(xc + y, yc - x, c);
        setPixel(xc - y, yc - x, c);

        x++;

        if (d > 0) {
            y--;
            d += 4 * (x - y) + 10;
        } else {
            d += 4 * x + 6;
        }
    }
}

void drawTriangle(
    int x1, int y1,
    int x2, int y2,
    int x3, int y3,
    char c)
{
    drawLine(x1, y1, x2, y2, c);
    drawLine(x2, y2, x3, y3, c);
    drawLine(x3, y3, x1, y1, c);
}

/* ============================
   Object Management
   ============================ */

void addObject(Shape shape) {
    if (objectCount < MAX_OBJECTS)
        objects[objectCount++] = shape;
}

void deleteObject(int index) {
    if (index < 0 || index >= objectCount)
        return;

    for (int i = index; i < objectCount - 1; i++)
        objects[i] = objects[i + 1];

    objectCount--;
}

void modifyObject(int index, Shape newShape) {
    if (index < 0 || index >= objectCount)
        return;

    objects[index] = newShape;
}

/* ============================
   Redraw Entire Scene
   ============================ */

void redrawScene() {
    clearCanvas();

    for (int i = 0; i < objectCount; i++) {

        Shape *s = &objects[i];

        switch (s->type) {

        case LINE:
            drawLine(
                s->x1, s->y1,
                s->x2, s->y2,
                s->drawChar);
            break;

        case RECTANGLE:
            drawRectangle(
                s->x1, s->y1,
                s->x2, s->y2,
                s->drawChar);
            break;

        case CIRCLE:
            drawCircle(
                s->x1, s->y1,
                s->radius,
                s->drawChar);
            break;

        case TRIANGLE:
            drawTriangle(
                s->x1, s->y1,
                s->x2, s->y2,
                s->x3, s->y3,
                s->drawChar);
            break;
        }
    }
}

/* ============================
   Demo
   ============================ */

int main() {

    Shape line = {
        .type = LINE,
        .x1 = 2, .y1 = 2,
        .x2 = 30, .y2 = 10,
        .drawChar = '*'
    };

    Shape rect = {
        .type = RECTANGLE,
        .x1 = 40, .y1 = 2,
        .x2 = 70, .y2 = 12,
        .drawChar = '#'
    };

    Shape circle = {
        .type = CIRCLE,
        .x1 = 20,
        .y1 = 15,
        .radius = 6,
        .drawChar = 'o'
    };

    Shape triangle = {
        .type = TRIANGLE,
        .x1 = 50, .y1 = 15,
        .x2 = 70, .y2 = 20,
        .x3 = 40, .y3 = 22,
        .drawChar = '+'
    };

    addObject(line);
    addObject(rect);
    addObject(circle);
    addObject(triangle);

    redrawScene();
    displayCanvas();

    printf("\nDeleting rectangle...\n\n");

    deleteObject(1);

    redrawScene();
    displayCanvas();

    return 0;
}
